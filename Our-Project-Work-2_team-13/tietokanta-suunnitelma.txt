### Komennot joilla luodaan tää tietokanta ###

mysql -u root -p

create database palaute;
use palaute;
create table vierailijat (
    id INT(200) PRIMARY KEY AUTO_INCREMENT,
    nimimerkki TEXT(25),
    sahkoposti VARCHAR(25),
    tahtoo TEXT(25),
    viesti VARCHAR(250)
);
